function MoveLinear(ned, Pstart, Pend, RPY, duration, nPoints) arguments ned Pstart (1,3) double Pend (1,3) double RPY (1,3) double duration = 3 nPoints = 50 end % Interpolation linéaire en XYZ x = linspace(Pstart(1), Pend(1), nPoints); y = linspace(Pstart(2), Pend(2), nPoints); z = linspace(Pstart(3), Pend(3), nPoints); curConf = [ned.CurrentConfiguration ned.GripperConfiguration]; traj = zeros(nPoints, ned.NumJoints); for i = 1:nPoints pos = [x(i) y(i) z(i)]; se = se3(flip(RPY),"eul","ZYX", pos); sol = ned.IK("tool_link", se.tform, ned.Weight, curConf); traj(i,:) = sol; curConf = sol; % warm start IK end % Exécuter la trajectoire ned.PlanTrajectory(traj, duration); end

function MoveCircular(ned, Pstart, Pvia, Pend, RPY, duration, nPoints) arguments ned Pstart (1,3) double Pvia (1,3) double Pend (1,3) double RPY (1,3) double duration = 3 nPoints = 60 end % --- Calcul du cercle passant par 3 points --- A = Pstart(:); B = Pvia(:); C = Pend(:); % Vecteurs du plan AB = B - A; AC = C - A; n = cross(AB, AC); n = n / norm(n); % Base locale du cercle ex = AB / norm(AB); ey = cross(n, ex); % Coordonnées locales xB = dot(AB, ex); yB = dot(AB, ey); xC = dot(AC, ex); yC = dot(AC, ey); % Centre du cercle en 2D d = 2*(xB*yC - yB*xC); if abs(d) < 1e-6 error("Points presque alignés → cercle impossible"); end ux = ( (xB^2 + yB^2)*yC - (xC^2 + yC^2)*yB ) / d; uy = ( (xC^2 + yC^2)*xB - (xB^2 + yB^2)*xC ) / d; % Centre 3D O = A + ux*ex + uy*ey; % Angles vA = A - O; vB = B - O; vC = C - O; thetaA = 0; thetaB = atan2(dot(vB,ey), dot(vB,ex)); thetaC = atan2(dot(vC,ey), dot(vC,ex)); % Choix du bon sens (A→B→C) if ~(thetaA < thetaB && thetaB < thetaC) && ... ~(thetaA > thetaB && thetaB > thetaC) if thetaC > thetaA thetaC = thetaC - 2*pi; else thetaC = thetaC + 2*pi; end end theta = linspace(thetaA, thetaC, nPoints); % Générer trajectoire cartésienne traj = zeros(nPoints,3); for i = 1:nPoints traj(i,:) = (O + cos(theta(i))*ex*norm(vA) + sin(theta(i))*ey*norm(vA))'; end % IK point par point curConf = [ned.CurrentConfiguration ned.GripperConfiguration]; jointTraj = zeros(nPoints, ned.NumJoints); for i = 1:nPoints se = se3(flip(RPY),"eul","ZYX", traj(i,:)); sol = ned.IK("tool_link", se.tform, ned.Weight, curConf); jointTraj(i,:) = sol; curConf = sol; end % Exécuter ned.PlanTrajectory(jointTraj, duration); end

Annexe 08

Définition de MoveIt

MoveIt est un framework ROS dédié à la manipulation robotique, dont le rôle principal est de transformer une intention de mouvement (ex. aller à cette position) en une trajectoire réalisable, sûre et conforme aux contraintes du robot.

Autrement dit :
  • tu ne commandes pas directement les moteurs,
  • tu décris un objectif,
  • MoveIt se charge de tout le raisonnement complexe.
Ce que MoveIt N'EST PAS

C'est important pour bien le situer.
MoveIt n'est pas :

  • Un simulateur (ce rôle revient à Gazebo)
  • Un driver matériel
  • Un remplacement de ROS
Le problème que MoveIt résout

Sans MoveIt, pour déplacer un robot manipulateur, il faudrait :

  • résoudre la cinématique inverse
  • Gérer les limites articulaires
  • éviter les collisions
  • interpoler une trajectoire continue
  • respecter les contraintes mécaniques
  • gérer les singularités
  • vérifier que le mouvement est physiquement réalisable

Tout cela est long, complexe, très sujet aux erreurs.

Donc MoveIt existe précisément pour abstraire cette complexité

Ce que fait MoveIt concrètement

On peut résumer MoveIt comme un pipeline de planification. Quand on écrit par exemple :

group.set_pose_target(pose) group.go()

MoveIt effectue en réalité les étapes suivantes :

      1. Interprétation de l'objectif
      • position cartésienne ou valeurs articulaires
      2. Calcul de la cinématique inverse
      • trouver des angles de joints compatibles
      3. Vérification des limites
      • positions, vitesses, accélérations
      4. Détection de collisions
      • robot / environnement / lui-même
      5. Planification de trajectoire
      • via des planificateurs (OMPL, RRT, PRM…)
      6. Génération d'une trajectoire temporelle
      7. Exécution ou simulation du mouvement

Tout cela est transparent pour l'utilisateur.

Les briques internes de MoveIt

MoveIt repose sur plusieurs composants clés :

  • URDF (modèle géométrique) : forme du robot, dimensions des liens, axes de rotation
  • SRDF (modèle sémantique) : groupes de joints ('arm'), end-effector, liens autorisés en collision
  • TF : gestion des repères ('base_link', 'tool_link', etc.)
  • Planning Scene : état courant du robot, obstacles, environnement
  • Planificateurs : OMPL (le plus courant), algorithmes probabilistes

#!/usr/bin/env python3 import sys import rospy import moveit_commander import geometry_msgs.msg from math import pi def simple_move(): # 1. INITIALISATION # ----------------- # On initialise le "Commander" MoveIt moveit_commander.roscpp_initialize(sys.argv) # On initialise le nœud ROS rospy.init_node('niryo_python_move', anonymous=True) # 2. DÉFINITION DES GROUPES # ------------------------- # Le robot est divisé en deux groupes de contrôle : # - Le bras (pour bouger) # - La pince (pour ouvrir/fermer) # Noms des groupes standards pour le Ned2 : 'arm' et 'tools' arm_group_name = "arm" gripper_group_name = "tools" move_group = moveit_commander.MoveGroupCommander(arm_group_name) gripper_group = moveit_commander.MoveGroupCommander(gripper_group_name) # Petit message pour dire qu'on est prêt print("=== Prêt à bouger le robot ! ===") # 3. MOUVEMENT 1 : Position Home Personnalisée # -------------------------------------------- print("1. Allons à la position Home (Définie manuellement)") # On définit les angles en radians pour une belle position d'attente # [J1, J2, J3, J4, J5, J6] # J5 = 0.0 -> Pour aligner la caméra avec le bras # J6 = 1.57 -> Pour tourner la pince de 90° (si besoin selon votre montage) home_joints = [0.0, 0.5, -1.2, 0.0, 0.0, 1.57] move_group.set_joint_value_target(home_joints) success = move_group.go(wait=True) move_group.stop() # Sécurité move_group.clear_pose_targets() if success: print(" -> Robot arrivé au Home !") else: print(" -> Échec du mouvement.") rospy.sleep(1) # 4. MOUVEMENT 2 : Simuler une approche (Aller vers le bas) # --------------------------------------------------------- print("2. Descente vers la zone de prise (Pick)") # On récupère la position actuelle pour la modifier current_pose = move_group.get_current_pose().pose # On définit une nouvelle cible (Target) target_pose = geometry_msgs.msg.Pose() target_pose.orientation = current_pose.orientation # On garde la même orientation target_pose.position.x = 0.25 # 25cm devant le robot target_pose.position.y = 0.0 # Centré target_pose.position.z = 0.15 # 15cm de hauteur (Z) move_group.set_pose_target(target_pose) # On bouge ! success = move_group.go(wait=True) move_group.stop() move_group.clear_pose_targets() print(" Mouvement réussi ?", success) rospy.sleep(1) # 5. ACTION PINCE : Fermer la pince # --------------------------------- print("3. Fermeture de la pince") # Pour le gripper, on peut utiliser des positions nommées comme "close" ou "open" # Note : Sur certains modèles Niryo, il faut parfois définir les joints manuellement try: gripper_group.set_named_target("close") gripper_group.go(wait=True) except: print(" (Pas de position 'close' trouvée, on continue sans pince)") rospy.sleep(1) # 6. MOUVEMENT 3 : Déposer ailleurs (Place) # ----------------------------------------- print("4. On déplace l'objet vers une autre zone") # ÉTAPE A : On lève l'objet (Sécurité) # ------------------------------------ # On récupère la position actuelle pour juste changer Z current_pose = move_group.get_current_pose().pose target_pose = current_pose target_pose.position.z = 0.30 # On monte à 30cm de haut move_group.set_pose_target(target_pose) move_group.go(wait=True) move_group.stop() move_group.clear_pose_targets() rospy.sleep(0.5) # ÉTAPE B : Rotation de la base (Joint 1) # ------------------------------------- print(" -> Rotation de la base à 90 degrés...") # On récupère les angles actuels des moteurs current_joints = move_group.get_current_joint_values() # On modifie juste le premier angle (Base / Joint 1) # 1.57 radians = environ 90 degrés vers la gauche # -1.57 radians = environ 90 degrés vers la droite current_joints[0] = 1.57 move_group.set_joint_value_target(current_joints) move_group.go(wait=True) move_group.stop() rospy.sleep(0.5) # ÉTAPE C : On pose (On redescend) # -------------------------------- print(" -> On pose l'objet") # On reprend la position actuelle (qui est maintenant tournée) target_pose = move_group.get_current_pose().pose target_pose.position.z = 0.15 # On redescend à la hauteur de la table move_group.set_pose_target(target_pose) move_group.go(wait=True) move_group.stop() move_group.clear_pose_targets() rospy.sleep(1) # 7. RETOUR HOME # -------------- print("5. Retour à la maison et ouverture pince") # 1. Ouvrir la pince # (Si "open" ne marche pas, on peut l'ignorer pour l'instant ou utiliser des joints) try: gripper_group.set_named_target("open") gripper_group.go(wait=True) except: print(" (La commande 'open' n'est pas connue, on continue)") # 2. Retourner à la position Home (MANUELLE) # On réutilise les mêmes valeurs que tout à l'heure # [Base, Épaule, Coude, Avant-bras, Poignet, Main] home_joints = [0.0, 0.5, -1.2, 0.0, 0.0, 1.57] move_group.set_joint_value_target(home_joints) success = move_group.go(wait=True) print("=== Terminé avec succès ! ===") # Arrêt propre de MoveIt moveit_commander.roscpp_shutdown()

APPROCHE (JOINTS) - Pince verticale print("--- Approche ---") # J5 = -1.57 (Pince vers le bas) approach_joints = [0.0, 0.0, -0.5, 0.0, -0.8, 0.0] arm_group.set_joint_value_target(approach_joints) success = arm_group.go(wait=True) if not success: print("Echec approche.") return # 2. OUVERTURE if gripper_group: gripper_group.set_named_target("open") gripper_group.go(wait=True) # 3. DESCENTE (CARTESIEN) print("--- Descente ---") current_pose = arm_group.get_current_pose().pose grasp_pose = current_pose grasp_pose.position.z = 0.055 # Hauteur de saisie (plan, fraction) = arm_group.compute_cartesian_path([grasp_pose], 0.01) if fraction > 0.9: arm_group.execute(plan, wait=True) else: print("Plan cartésien échoué, utilisation standard") arm_group.set_pose_target(grasp_pose) arm_group.go(wait=True) # 4. FERMETURE (AVEC GESTION D'ERREUR) print("--- Fermeture ---") if gripper_group: try: # On lance la fermeture. # MoveIt va peut-être crier "TOLERANCE VIOLATED" car le cube bloque. # On s'en fiche, tant que c'est fermé. gripper_group.set_named_target("close") gripper_group.go(wait=True) except: print(">>> Pince bloquée par le cube (Succès !)") # Important : On force l'arrêt du moteur de la pince pour ne pas surchauffer (virtuellement) gripper_group.stop() rospy.sleep(1.0) # 5. Méthodologie de mise en place d'un tunnel SSH

Prérequis

• SSH accessible sur robot :
On branche le câble Ethernet entre le robot et le PC. Puis on vérifie la connexion :

ping 169.254.200.200 # Ctrl+C pour arrêter

• Si la connexion est bien faite, on peut se connecter au terminal SSH au robot :

ssh niryo@169.254.200.200 # Password : robotics

Si la connexion est réussie, vous devriez voir ceci en début de ligne :

Figure 126 : Début de ligne du terminal SSH (robot)

Créer le tunnel SSH

• Nouveau terminal mais pas SSH, ici c'est celui de l'environnement Ubuntu qu'on veut :

Figure 127 : Début de ligne du terminal du PC

• Saisir la commande :

ssh -f -N -L 11311:localhost:11311 niryo@169.254.200.200

Les explications de cette commande sont les suivantes :

ArgumentRôle

Configurer ROS sur le robot (à ne pas faire avec le Ned2)

Dans un ROS multi-machine classique :

  • chaque machine est un nœud ROS
  • chacune doit connaître : le ROS Master et son IP réseau

Mais dans notre cas :

  • le Ned2 refuse le multi-machine
  • il fonctionne en ROS localhost-only
  • le tunnel SSH est un trucage réseau, pas une vraie connexion ROS

On ferait des exports sur le robot uniquement si :

  • le robot acceptait ROS multi-machine
  • le port 11311 était ouvert sur le réseau
  • le PC était un vrai nœud ROS distant

Points d'attention

Le tunnel SSH doit rester ouvert. C'est pourquoi il est recommandé d'utiliser la commande :

Pour pouvoir le mettre en tâche de fond, il est recommandé de mettre « -f ».

#!/usr/bin/env python3 import sys import rospy import moveit_commander from niryo_tools import get_niryo_joints from niryo_robot_rpi.srv import GetDigitalIO from niryo_robot_programs_manager.srv import SetConveyor def control_conveyor(conveyor_id, action, speed=50): """ Active ou arrête le convoyeur """ service_name = '/niryo_robot_programs_manager/control_conveyor' rospy.wait_for_service(service_name) try: stop_conveyor = rospy.ServiceProxy(service_name, SetConveyor) # action : True pour lancer, False pour arrêter stop_conveyor(conveyor_id, action, speed, 1) # 1 = direction par défaut except rospy.ServiceException as e: rospy.logerr(f"Erreur convoyeur : {e}") def wait_for_ir_sensor(pin_id): """ Boucle d'attente jusqu'à détection (passage à 0) """ service_name = '/niryo_robot_rpi/get_digital_io' rospy.wait_for_service(service_name) get_io = rospy.ServiceProxy(service_name, GetDigitalIO) rospy.loginfo(f"Attente détection sur le port {pin_id}...") while not rospy.is_shutdown(): res = get_io(pin_id) if res.value == 0: # Détection (logique inverse) rospy.loginfo("Objet détecté !") break rospy.sleep(0.1) # Évite de surcharger le processeur def main(): moveit_commander.roscpp_initialize(sys.argv) rospy.init_node("conveyor_ir_control", anonymous=True) group = moveit_commander.MoveGroupCommander("arm") # ID du convoyeur (souvent 1 ou 2) et Port du capteur (DI5) CONVEYOR_ID = 9 SENSOR_PIN = "DI5" # 1. Lancer le convoyeur rospy.loginfo("Lancement du convoyeur...") control_conveyor(CONVEYOR_ID, True, speed=30) # 2. Attendre que le capteur détecte l'objet wait_for_ir_sensor(SENSOR_PIN) # 3. Arrêter le convoyeur immédiatement control_conveyor(CONVEYOR_ID, False) rospy.loginfo("Convoyeur arrêté.") # 4. Déplacement du robot vers pos_cam pour la saisie ou l'inspection joints_cam = get_niryo_joints("pos_cam") if joints_cam: rospy.loginfo("Déplacement vers pos_cam...") group.go(joints_cam, wait=True) group.stop() rospy.loginfo("Robot en position.") moveit_commander.roscpp_shutdown() if __name__ == "__main__": main()
